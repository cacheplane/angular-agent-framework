# AG-UI Itinerary Redesign — Phase 2: App Mode + Google Map — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Prerequisite:** Phase 1 ([2026-06-22-ag-ui-itinerary-phase-1-productize.md](2026-06-22-ag-ui-itinerary-phase-1-productize.md)) is merged to `main`.

**Goal:** Add an "App mode" toolbar toggle that switches the AG-UI demo into a Tesla-style natural-language map cockpit — full-bleed Google Map on the left, chat copilot on the right, the Phase-1 panel reused as a floating overlay.

**Architecture:** New `MapCanvasComponent` using `@angular/google-maps` with a custom dark style. Stops gain optional `lat?`/`lng?` coordinates populated by a new `GeocodingService` invoked at the client-tool seam (agent stays unaware). App-mode signal lives on `AgUiShell`, persisted via `PalettePersistence`, drives a layout switch on the shell template.

**Tech Stack:** Angular 21.1 (zoneless), `@angular/google-maps` 21, Google Maps JavaScript API + Geocoding API (single `GOOGLE_MAPS_API_KEY`), Vitest, Playwright.

**Reference:** [docs/superpowers/specs/2026-06-22-ag-ui-itinerary-redesign-design.md](../specs/2026-06-22-ag-ui-itinerary-redesign-design.md) — Phase 2 sections.

```js
// SPDX-License-Identifier: MIT
import { readFileSync, writeFileSync, existsSync } from 'node:fs';
import { resolve, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const repoRoot = resolve(__dirname, '../../../..');

function readDotEnv() {
  const envPath = resolve(repoRoot, '.env');
  if (!existsSync(envPath)) return {};
  const raw = readFileSync(envPath, 'utf8');
  const out = {};
  for (const line of raw.split('\n')) {
    const m = line.match(/^([A-Z_][A-Z0-9_]*)=(.*)$/);
    if (m) out[m[1]] = m[2].replace(/^"|"$/g, '');
  }
  return out;
}

const env = { ...readDotEnv(), ...process.env };
const key = env.GOOGLE_MAPS_API_KEY ?? '';

const targetPath = resolve(__dirname, '../src/environments/generated-keys.local.ts');
const contents = `// SPDX-License-Identifier: MIT
// AUTO-GENERATED by scripts/inject-env.mjs. Do not edit by hand.
export const GENERATED_KEYS = {
  googleMaps: ${JSON.stringify(key)},
} as const;
`;
writeFileSync(targetPath, contents);
console.log(`[inject-env] wrote generated-keys.local.ts (key length: ${key.length})`);
```

```ts
// SPDX-License-Identifier: MIT
import { Injectable } from '@angular/core';

declare const google: {
  maps?: {
    Geocoder: new () => {
      geocode(req: { address: string }): Promise<{
        results: Array<{ geometry: { location: { lat(): number; lng(): number } } }>;
      }>;
    };
  };
};

@Injectable({ providedIn: 'root' })
export class GeocodingService {
  private geocoder: InstanceType<typeof google.maps.Geocoder> | null = null;

  async geocode(address: string): Promise<{ lat: number; lng: number } | null> {
    const g = (globalThis as { google?: typeof google }).google;
    if (!g?.maps?.Geocoder) return null;
    if (!this.geocoder) this.geocoder = new g.maps.Geocoder();
    try {
      const { results } = await this.geocoder.geocode({ address });
      const first = results[0];
      if (!first) return null;
      return { lat: first.geometry.location.lat(), lng: first.geometry.location.lng() };
    } catch {
      return null;
    }
  }
}
```

```ts
import { GeocodingService } from './geocoding.service';

export function itineraryClientTools(): ClientToolRegistry {
  const store = inject(ItineraryStore);
  const geocoding = inject(GeocodingService);

  return tools({
    // ...existing get_itinerary unchanged...

    add_stop: action(
      'Add a stop to a day of the trip itinerary. Afterwards, show the updated day with day_card.',
      z.object({ day: z.number().int().min(1), place: z.string(), note: z.string().optional() }),
      async ({ day, place, note }) => {
        const coords = (await geocoding.geocode(place)) ?? undefined;
        return { added: store.add(day, place, note, coords ? { coords } : undefined) };
      },
    ),

    move_stop: action(
      'Move an existing stop (matched by place name) to another day. Afterwards, show the updated day with day_card.',
      z.object({ place: z.string(), toDay: z.number().int().min(1) }),
      async ({ place, toDay }) => {
        const target = store.stops().find((s) => s.place.toLowerCase() === place.toLowerCase());
        if (!target) {
          return { error: `No stop named "${place}" — call get_itinerary to see what exists.` };
        }
        const toDayStops = store.stops().filter((s) => s.day === toDay && s.id !== target.id);
        store.reorder(target.id, toDay, toDayStops.length);
        return { moved: { ...target, day: toDay } };
      },
    ),

    // ...reorder_stop, clear_day, day_card unchanged...
  });
}
```

Note: `move_stop` does NOT re-geocode — coordinates travel with the stop unchanged.

```ts
const PARIS_CENTER: google.maps.LatLngLiteral = { lat: 48.8566, lng: 2.3522 };

@Component({
  selector: 'app-map-canvas',
  standalone: true,
  imports: [GoogleMap, MapMarker, MapPolyline, MapInfoWindow],
  changeDetection: ChangeDetectionStrategy.OnPush,
  template: `
    <google-map
      width="100%"
      height="100%"
      [center]="center()"
      [zoom]="zoom()"
      [options]="mapOptions"
    >
      @for (s of stopsWithCoords(); track s.id) {
        <map-marker
          [position]="{ lat: s.lat!, lng: s.lng! }"
          [options]="markerOptions(s)"
          (mapClick)="onMarkerClick(s)"
        />
      }
      @for (line of polylines(); track line.day) {
        <map-polyline [path]="line.path" [options]="polylineOptions(line.day)" />
      }
    </google-map>
  `,
  styles: [`:host { display: block; width: 100%; height: 100%; }`],
})
export class MapCanvasComponent {
  protected readonly store = inject(ItineraryStore);
  protected readonly center = signal<google.maps.LatLngLiteral>(PARIS_CENTER);
  protected readonly zoom = signal<number>(12);
  protected readonly mapOptions: google.maps.MapOptions = {
    styles: DARK_STYLE,
    disableDefaultUI: true,
    zoomControl: true,
    clickableIcons: false,
  };

  protected readonly stopsWithCoords = computed(() =>
    this.store.stops().filter((s) => s.lat != null && s.lng != null),
  );

  protected readonly polylines = computed(() => {
    const byDay = new Map<number, ItineraryStop[]>();
    for (const s of this.stopsWithCoords()) {
      byDay.set(s.day, [...(byDay.get(s.day) ?? []), s]);
    }
    return [...byDay.entries()]
      .filter(([, stops]) => stops.length >= 2)
      .map(([day, stops]) => ({
        day,
        path: stops.map((s) => ({ lat: s.lat!, lng: s.lng! })),
      }));
  });

  protected markerOptions(s: ItineraryStop): google.maps.MarkerOptions {
    return {
      icon: {
        path: 0, // google.maps.SymbolPath.CIRCLE
        fillColor: DAY_COLORS[(s.day - 1) % DAY_COLORS.length],
        fillOpacity: 1,
        strokeColor: '#fff',
        strokeWeight: 2,
        scale: 8,
      },
      title: s.place,
    };
  }

  protected polylineOptions(day: number): google.maps.PolylineOptions {
    return {
      strokeColor: DAY_COLORS[(day - 1) % DAY_COLORS.length],
      strokeOpacity: 0.7,
      strokeWeight: 3,
    };
  }

  protected onMarkerClick(s: ItineraryStop): void {
    this.store.focus(s.id);
  }
}
```

Add the class fields:

```ts
private readonly infoWindow = viewChild(MapInfoWindow);
private readonly markers = viewChildren(MapMarker);
protected readonly focused = computed(() =>
  this.store.stops().find((s) => s.id === this.store.focusedStopId()) ?? null,
);

constructor() {
  effect(() => {
    const focused = this.focused();
    if (!focused) return;
    const targetIdx = this.stopsWithCoords().findIndex((s) => s.id === focused.id);
    if (targetIdx < 0) return;
    const marker = this.markers()[targetIdx];
    const window = this.infoWindow();
    if (marker && window) {
      window.open(marker);
      this.center.set({ lat: focused.lat!, lng: focused.lng! });
    }
  });
}

protected onMarkerClick(s: ItineraryStop, _marker: MapMarker): void {
  this.store.focus(s.id);
}

protected removeFocused(): void {
  const f = this.focused();
  if (!f) return;
  this.store.remove(f.id, { source: 'user' });
  this.store.focus(null);
  this.infoWindow()?.close();
}
```
